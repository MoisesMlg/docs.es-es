```xml
<?xml version="1.0"?>
  <books>
    <book>
        <author>Carson</author>
        <price format="dollar">31.95</price>
        <pubdate>05/01/2001</pubdate>

        <publisher>MSPress</publisher>
        <state>WA</state>
    </pubinfo>
```
